CREATE TABLE XXBCM_SUPPLIER_CONTACT
(
	SUPPLIER_CONTACT_ID NUMBER GENERATED BY DEFAULT AS IDENTITY,	
	SUPP_CONTACT_NAME 		VARCHAR2(50) NOT NULL,
	SUPP_CONTACT_NUMBER		VARCHAR2(20) NOT NULL,
	SUPP_CONTACT_NUMBER_2  	VARCHAR2(20),
	PRIMARY KEY(SUPPLIER_CONTACT_ID)
);

CREATE TABLE XXBCM_SUPPLIER
   (
	SUPPLIER_ID				NUMBER GENERATED BY DEFAULT AS IDENTITY,
	SUPPLIER_NAME 			VARCHAR2(50) NOT NULL, 
	SUPP_ADDRESS 			VARCHAR2(100) NOT NULL,  
	SUPP_EMAIL 				VARCHAR2(60) NOT NULL,
	SUPP_CONTACT 			NUMBER,
	PRIMARY KEY(SUPPLIER_ID),
	FOREIGN KEY(SUPP_CONTACT) REFERENCES XXBCM_SUPPLIER_CONTACT(SUPPLIER_CONTACT_ID)
	);				

CREATE TABLE XXBCM_INVOICE_STATUS
(
	INVOICE_STATUS_ID NUMBER(3) GENERATED BY DEFAULT AS IDENTITY,
	DESCRIPTION VARCHAR2(100),
	PRIMARY KEY(INVOICE_STATUS_ID)
);

CREATE TABLE XXBCM_ORDER_STATUS
(
	ORDER_STATUS_ID NUMBER(3) GENERATED BY DEFAULT AS IDENTITY,
	DESCRIPTION VARCHAR2(100) NOT NULL,
	PRIMARY KEY(ORDER_STATUS_ID)
);

CREATE TABLE XXBCM_ORDER
(
	ORDER_ID				NUMBER GENERATED BY DEFAULT AS IDENTITY,
	SUPPLIER_ID				NUMBER NOT NULL, 
	ORDER_REF				VARCHAR2(20) NOT NULL,
	ORDER_DATE				DATE NOT NULL, 
	ORDER_TOTAL_AMOUNT		NUMBER(7,2),
	ORDER_DESCRIPTION		VARCHAR2(100) NOT NULL, 
	ORDER_STATUS			NUMBER NOT NULL, 
	ORDER_LINE_AMOUNT		NUMBER(7,2),
	PRIMARY KEY(ORDER_ID),
	FOREIGN KEY(SUPPLIER_ID) REFERENCES XXBCM_SUPPLIER(SUPPLIER_ID),
	FOREIGN KEY(ORDER_STATUS) REFERENCES XXBCM_ORDER_STATUS(ORDER_STATUS_ID)
);

	CREATE TABLE XXBCM_INVOICE
(
	INVOICE_ID				NUMBER GENERATED BY DEFAULT AS IDENTITY,
	ORDER_ID				NUMBER NOT NULL,
	INVOICE_REFERENCE		VARCHAR2(50),
	INVOICE_DATE			DATE,
	INVOICE_STATUS			NUMBER NOT NULL,
	INVOICE_HOLD_REASON		VARCHAR2(50),
	INVOICE_AMOUNT			NUMBER(7,2),
	INVOICE_DESCRIPTION		VARCHAR2(100),
	PRIMARY KEY(INVOICE_ID),
	FOREIGN KEY(ORDER_ID) REFERENCES XXBCM_ORDER(ORDER_ID),
	FOREIGN KEY(INVOICE_STATUS) REFERENCES XXBCM_INVOICE_STATUS(INVOICE_STATUS_ID)
);
